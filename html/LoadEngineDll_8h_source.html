49 inline FARPROC LoadEngineDll(const char* EngineName, const char* GetFactoryFuncName)
50 {
51  const size_t StringBufferSize = 4096;
52  char* LibName = (char*)malloc(StringBufferSize);
53  FARPROC GetFactoryFunc = NULL;
54  HMODULE hModule = NULL;
55 
56  // clang-format off
57 #if _WIN64
58  const char* Arch = "_64";
59 #else
60  const char* Arch = "_32";
61 #endif
62 
63 #ifdef _DEBUG
64  const char* Conf = "d";
65 #else
66  const char* Conf = "r";
67 #endif
68 
69  sprintf_s(LibName, StringBufferSize, "%s%s%s.dll", EngineName, Arch, Conf);
70 
71 #if PLATFORM_WIN32
72  hModule = LoadLibraryA(LibName);
73 #elif PLATFORM_UNIVERSAL_WINDOWS
74  hModule = LoadPackagedLibrary(WidenString(LibName).c_str(), 0);
75 #else
76 # error Unexpected platform
77 #endif
78  // clang-format on
79 
80  if (hModule == NULL)
81  {
82  printf("Failed to load %s library.\n", LibName);
83  OutputDebugStringA("Failed to load engine DLL");
84  free(LibName);
85  return NULL;
86  }
87 
88  GetFactoryFunc = GetProcAddress(hModule, GetFactoryFuncName);
89  if (GetFactoryFunc == NULL)
90  {
91  printf("Failed to load %s function from %s library.\n", GetFactoryFuncName, LibName);
92  OutputDebugStringA("Failed to load engine factory function from library");
93  FreeLibrary(hModule);
94  }
95 
96  free(LibName);
97  return GetFactoryFunc;
98 }
