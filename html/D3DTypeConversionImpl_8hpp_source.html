40 template <typename D3D_COMPARISON_FUNC>
41 inline D3D_COMPARISON_FUNC ComparisonFuncToD3DComparisonFunc(COMPARISON_FUNCTION Func)
42 {
43  // D3D12_COMPARISON_FUNC is equal to D3D11_COMPARISON_FUNC
44  switch (Func)
45  {
46  // clang-format off
47  case COMPARISON_FUNC_UNKNOWN: UNEXPECTED("Comparison function is not specified" ); return D3D_COMPARISON_FUNC_ALWAYS;
48  case COMPARISON_FUNC_NEVER: return D3D_COMPARISON_FUNC_NEVER;
49  case COMPARISON_FUNC_LESS: return D3D_COMPARISON_FUNC_LESS;
50  case COMPARISON_FUNC_EQUAL: return D3D_COMPARISON_FUNC_EQUAL;
51  case COMPARISON_FUNC_LESS_EQUAL: return D3D_COMPARISON_FUNC_LESS_EQUAL;
52  case COMPARISON_FUNC_GREATER: return D3D_COMPARISON_FUNC_GREATER;
53  case COMPARISON_FUNC_NOT_EQUAL: return D3D_COMPARISON_FUNC_NOT_EQUAL;
54  case COMPARISON_FUNC_GREATER_EQUAL: return D3D_COMPARISON_FUNC_GREATER_EQUAL;
55  case COMPARISON_FUNC_ALWAYS: return D3D_COMPARISON_FUNC_ALWAYS;
56  default: UNEXPECTED("Unknown comparison function" ); return D3D_COMPARISON_FUNC_ALWAYS;
57  // clang-format on
58  }

211 template <typename D3D_BLEND>

213 {
214  // D3D11_BLEND and D3D12_BLEND are identical

220  if (!bIsInit)
221  {
222  // In a multithreaded environment, several threads can potentially enter
223  // this block. This is not a problem since they will just initialize the
224  // memory with the same values more than once
225 
226  // clang-format off
227  D3DBlend[BLEND_FACTOR_ZERO ] = D3D_BLEND_ZERO;
228  D3DBlend[BLEND_FACTOR_ONE ] = D3D_BLEND_ONE;

232  D3DBlend[BLEND_FACTOR_INV_SRC_ALPHA ] = D3D_BLEND_INV_SRC_ALPHA;
233  D3DBlend[BLEND_FACTOR_DEST_ALPHA ] = D3D_BLEND_DEST_ALPHA;
234  D3DBlend[BLEND_FACTOR_INV_DEST_ALPHA ] = D3D_BLEND_INV_DEST_ALPHA;
235  D3DBlend[BLEND_FACTOR_DEST_COLOR ] = D3D_BLEND_DEST_COLOR;

256  UNEXPECTED("Incorrect blend factor (", bf, ")");

296 {
297  // D3D_BLEND_DESC and D3D11_BLEND_DESC structures are identical
298  d3d12BlendDesc.AlphaToCoverageEnable = BSDesc.AlphaToCoverageEnable ? TRUE : FALSE;

306 
307  DstRTDesc.SrcBlend = BlendFactorToD3DBlend<D3D_BLEND>(SrcRTDesc.SrcBlend);
